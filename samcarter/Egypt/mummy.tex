\newcommand{\AnkhC}[1][1]{%
\scalebox{#1}{%
\begin{tikzpicture}
  \begin{scope}[scale=0.08]
    \path [fill=ruby] (0,0) [bend left=75] to (0,1) [bend left=75] to cycle;
  \end{scope}
  \begin{scope}[yshift=-1.1,scale=#1]
    \node {\color{gold}\Ankh};
  \end{scope}
\end{tikzpicture}}}
